        // --- Simulator Logic ---
        function loadScenarioContext() {
            const val = document.getElementById('scenarioSelect').value;
            const data = scenarios[val];
            document.getElementById('dynamicContext').innerHTML = data.context;

            const btn = document.getElementById('triggerBtn');
            btn.innerHTML = `<i class="fas ${data.icon}"></i> ${data.btnText}`;

            // Reset logs
            document.getElementById('backendLogs').innerHTML = '<p style="text-align: center; color: var(--text-muted); margin-top: 40px; font-size: 14px;">Waiting for teacher interaction...</p>';
        }

        const sleep = ms => new Promise(r => setTimeout(r, ms));

        async function executeScenario() {
            const val = document.getElementById('scenarioSelect').value;
            const data = scenarios[val];
            const logsContainer = document.getElementById('backendLogs');

            // Disable button during run
            const btn = document.getElementById('triggerBtn');
            btn.disabled = true;
            btn.style.opacity = '0.5';

            logsContainer.innerHTML = '';

            for (let step of data.steps) {
                // Create Node (Waiting State)
                const logEl = document.createElement('div');
                logEl.className = 'log-item state-waiting';

                const timeStr = new Date().toLocaleTimeString('en-US', { hour12: false, hour: "numeric", minute: "numeric", second: "numeric" });

                logEl.innerHTML = `
                    <div class="log-icon"><i class="fas fa-circle"></i></div>
                    <div class="log-content">
                        <div class="log-title">${step.name}</div>
                        <div class="log-desc" id="desc-${Math.random()}">Initializing...</div>
                    </div>
                    <div class="badge">Waiting</div>
                `;

                logsContainer.appendChild(logEl);
                logsContainer.scrollTop = logsContainer.scrollHeight;

                await sleep(300); // brief pause before running

                // Transition to Running
                logEl.className = 'log-item state-running';
                logEl.querySelector('.log-icon i').className = 'fas fa-spinner';
                logEl.querySelector('.badge').innerText = 'Running';
                logEl.querySelector('.log-desc').innerText = 'Executing operation...';

                await sleep(step.time); // Simulate processing

                // Transition to Final State
                logEl.className = `log-item state-${step.status}`;

                let iconClass = 'fa-check';
                if (step.status === 'error') iconClass = 'fa-times';
                if (step.status === 'warning') iconClass = 'fa-exclamation';
